Project Setup

First, make sure you have a React project set up. You can create one using Create React App:
npx create-react-app css-in-js-example
cd css-in-js-example
npm install styled-components @emotion/react @emotion/styled

App.js

import React from 'react';
import styled from 'styled-components';
import { css } from '@emotion/react';
import styledEmotion from '@emotion/styled';
import styles from './App.module.css';

const App = () => {
  return (
    <div className="main-container">
      <Header>CSS-in-JS Examples</Header>

      {/* Introduction to CSS-in-JS */}
      <SectionHeader>Introduction to CSS-in-JS</SectionHeader>
      <p>
        CSS-in-JS is a technique where JavaScript is used to style components. It allows for scoped
        and dynamic styling, improving the maintainability and reusability of styles.
      </p>

      {/* Styled Components Example */}
      <SectionHeader>Styled Components</SectionHeader>
      <StyledButton>Styled Button</StyledButton>
      <p>
        Styled-components is a popular library for writing CSS-in-JS. It allows you to define styled
        components using template literals.
      </p>

      {/* Emotion Library Example */}
      <SectionHeader>Emotion Library</SectionHeader>
      <EmotionButton emotion={css`background-color: hotpink;`}>
        Emotion Button
      </EmotionButton>
      <p>
        Emotion is another library for writing CSS-in-JS. It provides powerful and flexible ways to
        style components.
      </p>

      {/* CSS Modules Example */}
      <SectionHeader>CSS Modules</SectionHeader>
      <button className={styles.moduleButton}>CSS Module Button</button>
      <p>
        CSS Modules allow you to scope your CSS by automatically creating unique class names. This
        avoids style conflicts.
      </p>

      {/* Best Practices for CSS-in-JS */}
      <SectionHeader>Best Practices for CSS-in-JS</SectionHeader>
      <ul>
        <li>Keep styles scoped to the component to avoid conflicts.</li>
        <li>Use dynamic styling for themes and responsive design.</li>
        <li>Organize styled components for better maintainability.</li>
        <li>Minimize the use of global styles to reduce complexity.</li>
      </ul>

      <Footer>Footer with CSS-in-JS Examples</Footer>
    </div>
  );
};

export default App;

// Styled Components
const Header = styled.header`
  width: 100%;
  height: 15vh;
  background-color: #007acc;
  color: #fff;
  display: flex;
  align-items: center;
  justify-content: center;
  font-size: 3vw;
  text-align: center;
  border-radius: 10px 10px 0 0;
`;

const SectionHeader = styled.h2`
  margin-top: 40px;
  margin-bottom: 20px;
  font-size: 24px;
  font-weight: bold;
  color: #007acc;
`;

const StyledButton = styled.button`
  background-color: #007acc;
  color: #fff;
  padding: 10px 20px;
  border: none;
  border-radius: 5px;
  cursor: pointer;
  margin-bottom: 20px;

  &:hover {
    background-color: #005f99;
  }
`;

// Emotion
const EmotionButton = styledEmotion.button`
  background-color: hotpink;
  color: #fff;
  padding: 10px 20px;
  border: none;
  border-radius: 5px;
  cursor: pointer;
  margin-bottom: 20px;

  &:hover {
    background-color: deeppink;
  }
`;

// Footer
const Footer = styled.footer`
  background-color: #007acc;
  color: #fff;
  text-align: center;
  padding: 20px;
  margin-top: 20px;
  border-radius: 0 0 10px 10px;
`;
