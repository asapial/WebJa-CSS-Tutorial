    <script>
        // Adding and Removing Classes
        const toggleClassButton = document.getElementById('toggleClassButton');
        const toggleText = document.getElementById('toggleText');

        toggleClassButton.addEventListener('click', () => {
            toggleText.classList.toggle('hidden');
        });

        // Manipulating Styles with JavaScript
        const changeStyleButton = document.getElementById('changeStyleButton');
        const colorBox = document.getElementById('colorBox');

        changeStyleButton.addEventListener('click', () => {
            colorBox.style.backgroundColor = '#e74c3c';
        });

        // CSS Transitions with JavaScript
        const transitionButton = document.getElementById('transitionButton');
        const transitionBox = document.getElementById('transitionBox');

        transitionButton.addEventListener('click', () => {
            transitionBox.style.transform = 'scale(1.5)';
            transitionBox.style.backgroundColor = '#e74c3c';
        });

        // CSS Variables and JavaScript
        const variableButton = document.getElementById('variableButton');
        const variableBox = document.getElementById('variableBox');

        variableButton.addEventListener('click', () => {
            document.documentElement.style.setProperty('--main-bg-color', '#e74c3c');
        });

        // Intersection Observer API
        const observerSections = document.querySelectorAll('.observer-section');

        const observer = new IntersectionObserver(entries => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                }
            });
        }, {
            threshold: 0.1
        });

        observerSections.forEach(section => {
            observer.observe(section);
        });
    </script>

  1. Adding and Removing Classes:
    • JavaScript toggles the .hidden class on a paragraph element.
    • Example: Clicking the button toggles the visibility of the paragraph.
  2. Manipulating Styles with JavaScript:
    • JavaScript changes the background color of a .box element.
    • Example: Clicking the button changes the background color of the box.
  3. CSS Transitions with JavaScript:
    • JavaScript applies CSS transitions to a .box element.
    • Example: Clicking the button scales the box and changes its background color.
  4. CSS Variables and JavaScript:
    • JavaScript updates a CSS variable value to change the background color of a .variable-box element.
    • Example: Clicking the button changes the value of --main-bg-color CSS variable.
  5. Intersection Observer API:
    • JavaScript uses the Intersection Observer API to detect when sections come into view and applies a .visible class to fade them in.
    • Example: Scrolling down reveals the sections with a fade-in effect.
